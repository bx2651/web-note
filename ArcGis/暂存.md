```
        require([
            "esri/Map",
            "esri/views/MapView"
        ], function (Map, MapView) {

            var map = new Map({
                basemap: "topo-vector" //创建一个新的地图底图
            });

            var view = new MapView({//创建一个视图，视图的主要作用是显示图层，弹出窗口和UI小部件，处理用户交互，并指定应将地图重点放在世界的哪一部分（即地图的“范围”）
                container: "viewDiv",
                map: map,
                center: [116.2317, 39.5427],//地图中央点的经纬度
                zoom: 9//比例尺的大小
            });

            var view = new SceneView({
                camera: {
                    position: [
                        -122, // lon
                        38, // lat
                        50000  // elevation in meters
                    ],
                    heading: 95, // direction the camera is looking
                    tilt: 65 // tilt of the camera relative to the ground
                }
            });

            view.goTo({                            // go to point with a custom animation duration
                center: [-114, 39],//跳转到地图的这个点
                duration: 5000//跳转的时间为5秒钟
            });


        });
    </script>

```

new MapView创建的是2D地图，new SceneView创建的是3D地图

```

const mapView = new MapView({          // Create MapView object
  map: myMap,
  container: "mapViewDiv"
});

const sceneView = new SceneView({      // Create SceneView object
  map: myMap,
  container: "sceneViewDiv"
});

```
